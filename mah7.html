import express from 'express'; import cors from 'cors'; import { Configuration, OpenAIApi } from "openai"; const app = express(); app.use(cors()); app.use(express.json()); const configuration = new Configuration({ apiKey: process.env.OPENAI_API_KEY, // chave segura no ambiente do Replit }); const openai = new OpenAIApi(configuration); app.post('/chat', async (req, res) => { try { const { message } = req.body; const completion = await openai.createChatCompletion({ model: "gpt-3.5-turbo", messages: [{role: "user", content: message}], max_tokens: 500, temperature: 0.7, }); res.json({ reply: completion.data.choices[0].message.content }); } catch (error) { console.error(error); res.status(500).json({ error: 'Erro na API' }); } }); const port = process.env.PORT || 3000; app.listen(port, () => { console.log(`Servidor rodando na porta ${port}`); });
